               break;

             if(last[i].increase) {

               break;
             }
           }
           if(increaseFirst && increaseLast) {
             if(increaseFirst * 2 <= increaseLast ){
    	   emit({increase:{first:increaseFirst,last:increaseLast,id:doc.id}});
             }
           } 
       }
